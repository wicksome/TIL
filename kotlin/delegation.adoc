= Delegation

[source, kotlin]
----
data class PersonInfo(
    val fName: String,
    val lName: String,
    val dob: LocalDate,
)
data class Person(
    val fName: String,
    val lName: String,
    val dob: LocalDate,
    val id: UUID
)
----

[source, kotlin]
----
data class PersonInfo(
    override val fName: String,
    override val lName: String,
    override val dob: LocalDate,
) : PersonDefinition

data class Person(
    private val data: PersonInfo,
    val id: UUID
) : PersonDefinition by data

interface PersonDefinition {
    val fName: String
    val lName: String
    val dob: LocalDate
}
----
