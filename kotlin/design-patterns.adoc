=== 전략(Strategy)

```kotlin
internal class DesignPatternTest {

    @Test
    fun main() {
        listOf(
            Context(Person()),
            Context(Car()),
            Context(Airplane()),
        ).forEach {
            it.move()
        }
    }
}


interface Strategy {
    fun logic()
}

class Person : Strategy {
    override fun logic() {
        println("walk")
    }
}

class Car : Strategy {
    override fun logic() {
        println("drive")
    }
}

class Airplane : Strategy {
    override fun logic() {
        println("fly")
    }
}

data class Context(
    private val strategy: Strategy,
) {
    fun move() {
        strategy.logic()
    }
}
```
