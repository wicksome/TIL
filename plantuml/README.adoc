== Example

=== 글꼴 지정

[plnatuml]
....
@startuml
skinparam defaultFontName "NanumGothic"

@enduml
....

=== footer 설정

[plnatuml]
....
@startuml
center footer © NAVER FINANCIAL Corp. All Rights Reserved.

@enduml
....

=== 외부 puml 파일 참조하기

[plantuml]
....
include::example-diagram.puml[]
....

=== 다이어그램 가운데 정렬하기

[plantuml,align="center"]
....
@startuml
A -> B
@enduml
....

=== group 배경 투명하게 하기

[plantuml]
.AS-IS
....
@startuml
box "actor1" #lightblue
  participant A
  participant B
end box
participant C

A -> B
group group1
  B -> C
  C --> B
end group
@enduml
....

[plantuml]
.TO-BE
....
@startuml
box "actor1" #lightblue
  participant A
  participant B
end box
participant C

A -> B
group #transparent group1
  B -> C
  C --> B
end group
@enduml
....

=== 이모지 추가

[plantuml]
....
@startuml
A -> B: <:lock:> 잠금
@enduml
....
